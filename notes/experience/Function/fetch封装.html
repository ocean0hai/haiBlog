// 用来处理resposne body中的 code
function responseCodeHandler<T>(
  response: BaseApiResponse<T>
): Promise<BaseApiResponse<T>> {
  const code = response.code || 0;
  let message = response.message;
  const codeArr = [0, 200];
  if (codeArr.includes(Number(code))) {
    return Promise.resolve(response);
  }
  const description = `[${code}]：${message}`;
  notification.error({
    message,
    description,
  });
  return Promise.reject(response);
}

// 请求处理函数
function responseHandle<T>(response: Response) {
  return Promise.resolve(response)
    .then((res) => resposneStateHandler(res))
    .then((res) => res?.json())
    .then((res) => responseCodeHandler<T>(res));
}

export async function get<T>(
  url: string,
  data?: { [key: string]: unknown }
): Promise<BaseApiResponse<T>> {
  const locationSearch = data && queryString.stringify(data);
  const finUrl = locationSearch ? `${url}?${locationSearch}` : url;
  return fetch(finUrl, {
    headers: {
      "Content-Type": "application/json",
      ...getAuthorization(),
    },
  })
    .then((response) => responseHandle<T>(response))
    .catch((ex) => Promise.reject(ex));
}

//post请求
export function post<T>(Url: string, data?: any): Promise<BaseApiResponse<T>> {
  return fetch(Url, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      ...getAuthorization(),
    },
    body: typeof data === "string" ? data : JSON.stringify(data),
  })
    .then((response) => responseHandle<T>(response))
    .catch((ex) => Promise.reject(ex));
}

//put请求
export async function put<T>(
  url: string,
  body: any
): Promise<BaseApiResponse<T>> {
  const response = await fetch(url, {
    method: "PUT",
    headers: {
      "Content-Type": "application/json",
      ...getAuthorization(),
    },
    body: JSON.stringify(body),
  });
  if (!response.ok) {
    throw new Error(response.statusText);
  }
  const data = await response.json();
  return data;
}
//delete请求
export async function delet<T>(
  url: string,
  data?: any
): Promise<BaseApiResponse<T>> {
  const response = await fetch(url, {
    method: "DELETE",
    headers: {
      "Content-Type": "application/json",
      ...getAuthorization(),
    },
    body: JSON.stringify(data),
  });
  if (!response.ok) {
    throw new Error(response.statusText);
  }
  const res = await response.json();
  return res;
}
